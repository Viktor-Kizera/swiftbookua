* –ï–∫—Ä–∞–Ω–æ–≤–∞–Ω—ñ —Å–ø–µ—Ü—ñ–∞–ª—å–Ω—ñ —Å–∏–º–≤–æ–ª–∏: `\0` \(–Ω—É–ª—å–æ–≤–∏–π —Å–∏–º–≤–æ–ª\), `\\` \(–∑–≤–æ—Ä–æ—Ç–Ω–∏–π —Å–ª–µ—à\), `\t` \(–≥–æ—Ä–∏–∑–æ–Ω—Ç–∞–ª—å–Ω–∏–π —Ç–∞–±\), `\n` \(–ø–µ—Ä–µ–Ω–µ—Å–µ–Ω–Ω—è —Ä—è–¥–∫–∞\), `\r` \(–ø–æ–≤–µ—Ä–Ω–µ–Ω–Ω—è –∫–∞—Ä–µ—Ç–∫–∏\), `\"` \(–ø–æ–¥–≤—ñ–π–Ω—ñ –ª–∞–ø–∫–∏\) —Ç–∞ `\'` \(–æ–¥–∏–Ω–∞—Ä–Ω—ñ –ª–∞–ø–∫–∏\)
* –î–æ–≤—ñ–ª—å–Ω–∏–π —é–Ω—ñ–∫–æ–¥–æ–≤–∏–π —Å–∫–∞–ª—è—Ä, –∑–∞–ø–∏—Å–∞–Ω–∏–π —É —Ñ–æ—Ä–º—ñ `\u{`n`}`, –¥–µ n - —Ü–µ 1‚Äì8-—Ü–∏—Ñ–æ–≤–µ —á–∏—Å–ª–æ –≤ —à—ñ—Å—Ç–Ω–∞–¥—Ü—è—Ç–∫–æ–≤–æ–º—É –∑–∞–ø–∏—Å—ñ –∑—ñ –∑–Ω–∞—á–µ–Ω–Ω—è–º, —â–æ –≤—ñ–¥–ø–æ–≤—ñ–¥–∞—î –¥—ñ–π—Å–Ω—ñ–π –∫–æ–¥–æ–≤—ñ–π –ø–æ–∑–∏—Ü—ñ—ó –Æ–Ω—ñ–∫–æ–¥—É.

–ö–æ–¥ –Ω–∏–∂—á–µ –ø–æ–∫–∞–∑—É—î —á–æ—Ç–∏—Ä–∏ –ø—Ä–∏–∫–ª–∞–¥–∏ —Å–ø–µ—Ü—ñ–∞–ª—å–Ω–∏—Ö —Å–∏–º–≤–æ–ª—ñ–≤. –ö–æ–Ω—Å—Ç–∞–Ω—Ç–∞ `wiseWords` –º—ñ—Å—Ç–∏—Ç—å –ø–∞—Ä—É –µ–∫—Ä–∞–Ω–æ–≤–∞–Ω–∏—Ö –ø–æ–¥–≤—ñ–π–Ω–∏—Ö –ª–∞–ø–æ–∫. –ö–æ–Ω—Å—Ç–∞–Ω—Ç–∏ `dollarSign`, `blackHeart`, —Ç–∞ `sparklingHeart` –¥–µ–º–æ–Ω—Å—Ç—Ä—É—é—Ç—å —Ñ–æ—Ä–º–∞—Ç —é–Ω—ñ–∫–æ–¥–æ–≤–∏—Ö —Å–∫–∞–ª—è—Ä—ñ–≤:

```swift
let wiseWords = "\"–£—è–≤–∞ –≤–∞–∂–ª–∏–≤—ñ—à–∞ –∑–∞ –∑–Ω–∞–Ω–Ω—è\" - –ï–π–Ω—à—Ç–µ–π–Ω"
// "–£—è–≤–∞ –≤–∞–∂–ª–∏–≤—ñ—à–∞ –∑–∞ –∑–Ω–∞–Ω–Ω—è" - –ï–π–Ω—à—Ç–µ–π–Ω
let dollarSign = "\u{24}"        // $,  Unicode scalar U+0024
let blackHeart = "\u{2665}"      // ‚ô•,  Unicode scalar U+2665
let sparklingHeart = "\u{1F496}" // üíñ, Unicode scalar U+1F496
```

–û—Å–∫—ñ–ª—å–∫–∏ –±–∞–≥–∞—Ç–æ—Ä—è–¥–∫–æ–≤—ñ –ª—ñ—Ç–µ—Ä–∞–ª–∏ –ø–∏—à—É—Ç—å—Å—è –∑ —Ç—Ä—å–æ–º–∞ –ø–∞—Ä–∞–º–∏ –ø–æ–¥–≤—ñ–π–Ω–∏—Ö –ª–∞–ø–æ–∫ –∑–∞–º—ñ—Å—Ç—å –ª–∏—à–µ –æ–¥–Ω—ñ—î—ó –ø–∞—Ä–∏, —Ç–æ –≤—Å–µ—Ä–µ–¥–∏–Ω—ñ —ó—Ö –º–æ–∂–Ω–∞ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É–≤–∞—Ç–∏ —Å–∏–º–≤–æ–ª –ª–∞–ø–æ–∫ (`"`) –±–µ–∑ –µ–∫—Ä–∞–Ω—É–≤–∞–Ω–Ω—è. –ê–ª–µ –¥–ª—è —Ç–æ–≥–æ, —â–æ–± –≤–∫–ª—é—á–∏—Ç–∏ —Ç–µ–∫—Å—Ç `"""` –¥–æ –±–∞–≥–∞—Ç–æ—Ä—è–¥–∫–æ–≤–æ–≥–æ –ª—ñ—Ç–µ—Ä–∞–ª—É, —Å–ª—ñ–¥ –µ–∫—Ä–∞–Ω—É–≤–∞—Ç–∏ —Ö–æ—á–∞ –± –æ–¥–Ω—É –ø–∞—Ä—É –ª–∞–ø–æ–∫. –ù–∞–ø—Ä–∏–∫–ª–∞–¥: 

```swift
let threeDoubleQuotationMarks = """
–ï–∫—Ä–∞–Ω—É–≤–∞–Ω–Ω—è –ø–µ—Ä—à–æ—ó –ø–∞—Ä–∏ –ª–∞–ø–æ–∫ \"""
–ï–∫—Ä–∞–Ω—É–≤–∞–Ω–Ω—è –≤—Å—ñ—Ö —Ç—Ä—å–æ—Ö –ø–∞—Ä –ª–∞–ø–æ–∫ \"\"\"
"""
```

### –†–æ–∑—à–∏—Ä–µ–Ω—ñ —Ä–æ–∑–¥—ñ–ª—é–≤–∞—á—ñ —Ä—è–¥–∫—ñ–≤

–†—è–¥–∫–æ–≤–∏–π –ª—ñ—Ç–µ—Ä–∞–ª –º–æ–∂–Ω–∞ —Ä–æ–∑–º—ñ—Å—Ç–∏—Ç–∏ –ø–æ–º—ñ–∂ —Ä–æ–∑—à–∏—Ä–µ–Ω–∏—Ö —Ä–æ–∑–¥—ñ–ª—é–≤–∞—á—ñ–≤: —Ç–∞–∫–∏–º —á–∏–Ω–æ–º –º–æ–∂–Ω–∞ –≤–∫–ª—é—á–∞—Ç–∏ –¥–æ —Ä—è–¥–∫–∞ —Å–ø–µ—Ü—ñ–∞–ª—å–Ω—ñ —Å–∏–º–≤–æ–ª–∏ –Ω–µ –≤–∏–∫–ª–∏–∫–∞—é—á–∏ —ó—Ö –µ—Ñ–µ–∫—Ç—ñ–≤. –î–ª—è —Ü—å–æ–≥–æ –ø–æ—Ç—Ä—ñ–±–Ω–æ —Ä–æ–∑–º—ñ—Å—Ç–∏—Ç–∏ —Å–≤—ñ–π —Ä—è–¥–æ–∫ –ø–æ–º—ñ–∂ –ª–∞–ø–æ–∫ (`"`) —Ç–∞ –æ—Ç–æ—á–∏—Ç–∏ —Ü–µ —Å–∏–º–≤–æ–ª–∞–º–∏ —Ä–µ—à—ñ—Ç–∫–∏ (`#`). –ù–∞–ø—Ä–∏–∫–ª–∞–¥, —è–∫—â–æ –º–∏ –Ω–∞–¥—Ä—É–∫—É—î–º–æ —Ä—è–¥–∫–æ–≤–∏–π –ª—ñ—Ç–µ—Ä–∞–ª `#"–†—è–¥–æ–∫ 1\n–†—è–¥–æ–∫ 2"# `, —Ç–æ –ø–æ–±–∞—á–∏–º–æ —Å–∏–º–≤–æ–ª –ø–µ—Ä–µ–Ω–µ—Å–µ–Ω–Ω—è —Ä—è–¥–∫—É `\n` –∑–∞–º—ñ—Å—Ç—å –¥—Ä—É–∫—É –ª—ñ—Ç–µ—Ä–∞–ª—É —É –¥–≤–∞ —Ä—è–¥–∫–∏. 

–Ø–∫—â–æ –∂ –ø–æ—Ç—Ä—ñ–±–Ω–æ, —â–æ–± –¥–µ—è–∫—ñ –∑—ñ —Å–ø–µ—Ü—ñ–∞–ª—å–Ω–∏—Ö —Å–∏–º–≤–æ–ª—ñ–≤ —É —Ä—è–¥–∫—É –¥—ñ—è–ª–∏, —Å–ª—ñ–¥ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É–≤–∞—Ç–∏ —Å–∏–º–≤–æ–ª –∞–±–æ —Å–∏–º–≤–æ–ª–∏ —Ä–µ—à—ñ—Ç–∫–∏ (`#`) —Ä–∞–∑–æ–º —ñ–∑ –±–µ–∫—Å–ª–µ—à–µ–º, —â–æ –µ–∫—Ä–∞–Ω—É—î —Å–ø–µ—Ü—ñ–∞–ª—å–Ω–∏–π —Å–∏–º–≤–æ–ª (`\`). –ü—Ä–∏ —Ü—å–æ–º—É –∫—ñ–ª—å–∫—ñ—Å—Ç—å —Å–∏–º–≤–æ–ª—ñ–≤ —Ä–µ—à—ñ—Ç–∫–∏ –ø—Ä–∏ –±–µ–∫—Å–ª–µ—à—ñ –º–∞—î –≤—ñ–¥–ø–æ–≤—ñ–¥–∞—Ç–∏ –∫—ñ–ª—å–∫–æ—Å—Ç—ñ —Å–∏–º–≤–æ–ª—ñ–≤ —Ä–µ—à—ñ—Ç–∫–∏ —Å–∞–º–æ–≥–æ –ª—ñ—Ç–µ—Ä–∞–ª–∞. –ù–∞–ø—Ä–∏–∫–ª–∞–¥, —è–∫—â–æ –º–∏ –º–∞—î–º–æ —Ä—è–¥–æ–∫ `#"Line 1\nLine 2"#` —ñ –Ω–∞–º –ø–æ—Ç—Ä—ñ–±–Ω–æ, —â–æ–± —Å–∏–º–≤–æ–ª –ø–µ—Ä–µ–Ω–µ—Å–µ–Ω–Ω—è —Ä—è–¥–∫—É ( `\n`) –ø—Ä–∞—Ü—é–≤–∞–≤, —Å–ª—ñ–¥ –Ω–∞—Ç–æ–º—ñ—Å—Ç—å –ø–∏—Å–∞—Ç–∏ —Ç–∞–∫: `#"Line 1\#nLine 2"#`. –ê–Ω–∞–ª–æ–≥—ñ—á–Ω–æ, —É –ª—ñ—Ç–µ—Ä–∞–ª—ñ `###"Line1\###nLine2"###` –ø–µ—Ä–µ–Ω–µ—Å–µ–Ω–Ω—è —Ä—è–¥–∫—É —Ç–∞–∫–æ–∂ –≤—ñ–¥–±—É–≤–∞—î—Ç—å—Å—è.

–†—è–¥–∫–æ–≤—ñ –ª—ñ—Ç–µ—Ä–∞–ª–∏, —è–∫—ñ —Å—Ç–≤–æ—Ä—é—é—Ç—å—Å—è –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é —Ä–æ–∑—à–∏—Ä–µ–Ω–∏—Ö —Ä–æ–∑–¥—ñ–ª—é–≤–∞—á—ñ–≤ —Ä—è–¥–∫—ñ–≤, –º–æ–∂—É—Ç—å –±—É—Ç–∏ —Ç–∞–∫–æ–∂ —ñ –±–∞–≥–∞—Ç–æ—Ä—è–¥–∫–æ–≤–∏–º–∏. –ó–æ–∫—Ä–µ–º–∞, –º–æ–∂–Ω–∞ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É–≤–∞—Ç–∏ —Ä–æ–∑—à–∏—Ä–µ–Ω—ñ —Ä–æ–∑–¥—ñ–ª—é–≤–∞—á—ñ –¥–ª—è —Ç–æ–≥–æ, —â–æ–± –≤–∫–ª—é—á–∏—Ç–∏ —Ç–µ–∫—Å—Ç `"""` –≤ –±–∞–≥–∞—Ç–æ—Ä—è–¥–∫–æ–≤–∏–π –ª—ñ—Ç–µ—Ä–∞–ª, —Ç–∞–∫–∏–º —á–∏–Ω–æ–º –∑–º—ñ–Ω—é—é—á–∏ –ø–æ–≤–µ–¥—ñ–Ω–∫—É —Ç–∞–∫–∏—Ö –ª—ñ—Ç–µ—Ä–∞–ª—ñ–≤ –∑–∞ –∑–∞–º–æ–≤—á–∞–Ω–Ω—è–º. –ù–∞–ø—Ä–∏–∫–ª–∞–¥:

```swift
let threeMoreDoubleQuotationMarks = #"""
–û—Å—å —â–µ –æ–¥–Ω–∞ —Ç—Ä—ñ–π–∫–∞ –ø–æ–¥–≤—ñ–π–Ω–∏—Ö –ª–∞–ø–æ–∫: """
"""#
```

## –Ü–Ω—ñ—Ü—ñ–∞–ª—ñ–∑–∞—Ü—ñ—è –ø–æ—Ä–æ–∂–Ω—å–æ–≥–æ —Ä—è–¥–∫–∞

–©–æ–± —Å—Ç–≤–æ—Ä–∏—Ç–∏ –ø–æ—Ä–æ–∂–Ω—ñ–π —Ä—è–¥–æ–∫, —è–∫ –ø–æ—á–∞—Ç–∫–æ–≤–∏–π –∫—Ä–æ–∫ –≤ –ø–æ–±—É–¥–æ–≤—ñ –¥–æ–≤—à–æ–≥–æ —Ä—è–¥–∫–∞, –º–æ–∂–Ω–∞ –∞–±–æ –ø—Ä–∏—Å–≤–æ—ó—Ç–∏ –ø–æ—Ä–æ–∂–Ω—ñ–π —Ä—è–¥–∫–æ–≤–∏–π –ª—ñ—Ç–µ—Ä–∞–ª –∑–º—ñ–Ω–Ω—ñ–π, –∞–±–æ —Å—Ç–≤–æ—Ä–∏—Ç–∏ –Ω–æ–≤–∏–π –µ–∫–∑–µ–º–ø–ª—è—Ä `String` –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é —Å–∏–Ω—Ç–∞–∫—Å–∏—Å—É —ñ–Ω—ñ—Ü—ñ–∞–ª—ñ–∑–∞—Ü—ñ—ó:

```swift
var emptyString = ""               // –ø–æ—Ä–æ–∂–Ω—ñ–π —Ä—è–¥–∫–æ–≤–∏–π –ª—ñ—Ç–µ—Ä–∞–ª
var anotherEmptyString = String()  // —Å–∏–Ω—Ç–∞–∫—Å–∏—Å —ñ–Ω—ñ—Ü—ñ–∞–ª—ñ–∑–∞—Ü—ñ—ó
// –æ–±–∏–¥–≤–∞ —Ä—è–¥–∫–∏ –ø–æ—Ä–æ–∂–Ω—ñ, —ñ –¥–æ—Ä—ñ–≤–Ω—é—é—Ç—å –æ–¥–Ω–µ –æ–¥–Ω–æ–º—É
```

–©–æ–± –≤–∏–∑–Ω–∞—á–∏—Ç–∏, —á–∏ –ø–æ—Ä–æ–∂–Ω—ñ–π —Ä—è–¥–æ–∫, —Å–ª—ñ–¥ –ø–µ—Ä–µ–≤—ñ—Ä–∏—Ç–∏ –ë—É–ª–µ–≤–µ –∑–Ω–∞—á–µ–Ω–Ω—è –≤–ª–∞—Å—Ç–∏–≤–æ—Å—Ç—ñ `isEmpty`:

```swift
if emptyString.isEmpty {
    print("Nothing to see here")
}
// –î—Ä—É–∫—É—î "Nothing to see here"
```

## –ó–º—ñ–Ω—é–≤–∞–Ω—ñ—Å—Ç—å —Ä—è–¥–∫—ñ–≤

–©–æ–± –≤–∫–∞–∑–∞—Ç–∏, —á–∏ –º–æ–∂–Ω–∞ –∑–º—ñ–Ω–∏—Ç–∏ \(–∞–±–æ _–º—É—Ç—É–≤–∞—Ç–∏_\) –ø–µ–≤–Ω–µ –∑–Ω–∞—á–µ–Ω–Ω—è `String`, —Å–ª—ñ–¥ –∞–±–æ –ø—Ä–∏—Å–≤–æ—ó—Ç–∏ —Ü–µ –∑–Ω–∞—á–µ–Ω–Ω—è –∑–º—ñ–Ω–Ω—ñ–π \(—Ç–æ–¥—ñ –π–æ–≥–æ –º–æ–∂–Ω–∞ –±—É–¥–µ –∑–º—ñ–Ω–∏—Ç–∏\), –∞–±–æ –ø—Ä–∏—Å–≤–æ—ó—Ç–∏ —Ü–µ –∑–Ω–∞—á–µ–Ω–Ω—è –∫–æ–Ω—Å—Ç–∞–Ω—Ç—ñ \(—Ç–æ–¥—ñ –π–æ–≥–æ –Ω–µ –º–æ–∂–Ω–∞ –±—É–¥–µ –∑–º—ñ–Ω–∏—Ç–∏\):

You indicate whether a particular `String` can be modified \(or _mutated_\) by assigning it to a variable \(in which case it can be modified\), or to a constant \(in which case it cannot be modified\):

```swift
var variableString = "Horse"
variableString += " and carriage"
// –∑–º—ñ–Ω–Ω–∞ variableString —Ç–µ–ø–µ—Ä –º–∞—î –∑–Ω–∞—á–µ–Ω–Ω—è "Horse and carriage"

let constantString = "Highlander"
constantString += " and another Highlander"
// —Ü–µ –ø—Ä–∏–∑–≤–µ–¥–µ –¥–æ –ø–æ–º–∏–ª–∫–∏ —á–∞—Å—É –∫–æ–º–ø—ñ–ª—è—Ü—ñ—ó - –∫–æ–Ω—Å—Ç–∞–Ω—Ç–Ω–∏–π —Ä—è–¥–æ–∫ –Ω–µ –º–æ–∂–Ω–∞ –∑–º—ñ–Ω—é–≤–∞—Ç–∏
```

> **–ü—Ä–∏–º—ñ—Ç–∫–∞**
>
> –î–∞–Ω–∏–π –ø—ñ–¥—Ö—ñ–¥ –≤—ñ–¥—Ä—ñ–∑–Ω—è—î—Ç—å—Å—è –≤—ñ–¥ –º—É—Ç–∞—Ü—ñ–π —Ä—è–¥–∫—ñ–≤ –≤ Objective-C —Ç–∞ Cocoa, –¥–µ –≤–∏ –æ–±–∏—Ä–∞—î—Ç–µ –º—ñ–∂ –¥–≤–æ–º–∞ –∫–ª–∞—Å–∞–º–∏, \(`NSString` —Ç–∞ `NSMutableString`\), —â–æ–± –≤–∫–∞–∑–∞—Ç–∏, —á–∏ –º–æ–∂–Ω–∞ –∑–º—ñ–Ω—é–≤–∞—Ç–∏ —Ä—è–¥–æ–∫.

## –†—è–¥–∫–∏ —î –¢–∏–ø–∞–º–∏-–∑–Ω–∞—á–µ–Ω–Ω—è–º–∏

–¢–∏–ø `String` —É Swift —î —Ç–∏–ø–æ–º-–∑–Ω–∞—á–µ–Ω–Ω—è–º. –Ø–∫—â–æ —Å—Ç–≤–æ—Ä–∏—Ç–∏ –Ω–æ–≤–µ –∑–Ω–∞—á–µ–Ω–Ω—è `String`, —Ü–µ –∑–Ω–∞—á–µ–Ω–Ω—è `String` –±—É–¥–µ _–∫–æ–ø—ñ—é–≤–∞—Ç–∏—Å—å_ –ø—Ä–∏ –ø–µ—Ä–µ–¥–∞–≤–∞–Ω–Ω—ñ –¥–æ —Ñ—É–Ω–∫—Ü—ñ—ó —á–∏ –º–µ—Ç–æ–¥—É, —á–∏ –ø—Ä–∏ –ø—Ä–∏—Å–≤–æ—î–Ω–Ω—ñ –∑–º—ñ–Ω–Ω—ñ–π —á–∏ –∫–æ–Ω—Å—Ç–∞–Ω—Ç—ñ. –í –∫–æ–∂–Ω–æ–º—É –≤–∏–ø–∞–¥–∫—É –±—É–¥–µ —Å—Ç–≤–æ—Ä—é–≤–∞—Ç–∏—Å—å –Ω–æ–≤–∞ –∫–æ–ø—ñ—è –Ω–∞—è–≤–Ω–æ–≥–æ –∑–Ω–∞—á–µ–Ω–Ω—è `String`, —ñ —Ü—è –Ω–æ–≤–∞ –∫–æ–ø—ñ—è –±—É–¥–µ –ø–µ—Ä–µ–¥–∞–≤–∞—Ç–∏—Å—å —á–∏ –ø—Ä–∏—Å–≤–æ—é–≤–∞—Ç–∏—Å—å, –∞ –Ω–µ –æ—Ä–∏–≥—ñ–Ω–∞–ª—å–Ω–∞ –≤–µ—Ä—Å—ñ—è. –¢–∏–ø–∏-–∑–Ω–∞—á–µ–Ω–Ω—è –¥–µ—Ç–∞–ª—å–Ω—ñ—à–µ –æ–ø–∏—Å–∞–Ω—ñ –≤ —Ä–æ–∑–¥—ñ–ª—ñ [–°—Ç—Ä—É–∫—Ç—É—Ä–∏ —Ç–∞ –ü–µ—Ä–µ—á–∏—Å–ª–µ–Ω–Ω—è —è–∫ –¢–∏–ø–∏-–∑–Ω–∞—á–µ–Ω–Ω—è]({% link _book/1_language_guide/8_classes_and_structures.md %}#—Å—Ç—Ä—É–∫—Ç—É—Ä–∏-—Ç–∞-–ø–µ—Ä–µ—á–∏—Å–ª–µ–Ω–Ω—è-—è–∫-—Ç–∏–ø–∏-–∑–Ω–∞—á–µ–Ω–Ω—è).

–ü–æ–≤–µ–¥—ñ–Ω–∫–∞ –∫–æ–ø—ñ—é–≤–∞–Ω–Ω—è –∑–∞ –∑–∞–º–æ–≤—á–∞–Ω–Ω—è–º —Ä—è–¥–∫—ñ–≤ —É Swift –≥–∞—Ä–∞–Ω—Ç—É—î, —â–æ –∫–æ–ª–∏ —Ñ—É–Ω–∫—Ü—ñ—è —á–∏ –º–µ—Ç–æ–¥ –ø–µ—Ä–µ–¥–∞—î –≤–∞–º –∑–Ω–∞—á–µ–Ω–Ω—è `String`, —Å–∞–º–µ –≤–∏ –≤–æ–ª–æ–¥—ñ—î—Ç–µ —Ü–∏–º —Ç–æ—á–Ω–∏–º –∑–Ω–∞—á–µ–Ω–Ω—è–º `String`, –Ω–µ–∑–∞–ª–µ–∂–Ω–æ –≤—ñ–¥ —Ç–æ–≥–æ, –∑–≤—ñ–¥–∫–∏ –≤–æ–Ω–æ –ø—Ä–∏–π—à–ª–æ. –í–∏ –º–æ–∂–µ—Ç–µ –±—É—Ç–∏ –≤–ø–µ–≤–Ω–µ–Ω—ñ –≤ —Ç–æ–º—É, —â–æ –ø–µ—Ä–µ–¥–∞–Ω–∏–π –≤–∞–º —Ä—è–¥–æ–∫ –Ω–µ –±—É–¥–µ –∑–º—ñ–Ω–µ–Ω–æ –Ω—ñ–∫–∏–º –æ–∫—Ä—ñ–º —è–∫ –≤–∞–º–∏ —Å–∞–º–∏–º–∏.

–ó–∞ –ª–∞—à—Ç—É–Ω–∫–∞–º–∏, –∫–æ–º–ø—ñ–ª—è—Ç–æ—Ä Swift –æ–ø—Ç–∏–º—ñ–∑–æ–≤—É—î –≤–∏–∫–æ—Ä–∏—Å—Ç–∞–Ω–Ω—è —Ä—è–¥–∫—ñ–≤ —Ç–∞–∫–∏–º —á–∏–Ω–æ–º, —â–æ —Ñ–∞–∫—Ç–∏—á–Ω–µ –∫–æ–ø—ñ—é–≤–∞–Ω–Ω—è –≤—ñ–¥–±—É–≤–∞—î—Ç—å—Å—è —Ç—ñ–ª—å–∫–∏ —Ç–æ–¥—ñ, –∫–æ–ª–∏ —Ü–µ –¥—ñ–π—Å–Ω–æ –Ω–µ–æ–±—Ö—ñ–¥–Ω–æ. –¶–µ –æ–∑–Ω–∞—á–∞—î, —â–æ –≤–∏ –∑–∞–≤–∂–¥–∏ –æ—Ç—Ä–∏–º–∞—î—Ç–µ —á—É–¥–æ–≤—É —à–≤–∏–¥–∫–æ–¥—ñ—é –ø—Ä–∏ —Ä–æ–±–æ—Ç—ñ –∑ —Ä—è–¥–∫–∞–º–∏ —è–∫ —ñ–∑ —Ç–∏–ø–∞–º–∏-–∑–Ω–∞—á–µ–Ω–Ω—è–º–∏.

## –†–æ–±–æ—Ç–∞ —ñ–∑ —Å–∏–º–≤–æ–ª–∞–º–∏

–û—Ç—Ä–∏–º–∞—Ç–∏ –¥–æ—Å—Ç—É–ø –¥–æ –æ–∫—Ä–µ–º–∏—Ö —Å–∏–º–≤–æ–ª—ñ–≤ `Character` —É —Ä—è–¥–∫—É `String` –º–æ–∂–Ω–∞ —ñ—Ç–µ—Ä—É—é—á–∏ –π–æ–≥–æ –≤–ª–∞—Å—Ç–∏–≤—ñ—Å—Ç—å `characters` –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é —Ü–∏–∫–ª—É `for`-`in`:

```swift
for character in "–°–æ–±–∞–∫–∞!üê∂".characters {
    print(character)
}
// –°
// –æ
// –±
// –∞
// –∫
// –∞
// !
// üê∂
```

–¶–∏–∫–ª `for`-`in` –¥–µ—Ç–∞–ª—å–Ω–æ –æ–ø–∏—Å–∞–Ω–∏–π –≤ –ø—ñ–¥—Ä–æ–∑–¥—ñ–ª—ñ [–¶–∏–∫–ª For-In]({% link _book/1_language_guide/4_control_flow.md %}#—Ü–∏–∫–ª-for-in).

–Ø–∫ –≤–∞—Ä—ñ–∞–Ω—Ç, –º–æ–∂–Ω–∞ —Å—Ç–≤–æ—Ä–∏—Ç–∏ –æ–∫—Ä–µ–º–∏–π —Å–∏–º–≤–æ–ª `Character` —É –≤–∏–≥–ª—è–¥—ñ –∫–æ–Ω—Å—Ç–∞–Ω—Ç–∏ —á–∏ –∑–º—ñ–Ω–Ω–æ—ó –∑ –æ–¥–Ω–æ—Å–∏–º–≤–æ–ª—å–Ω–æ–≥–æ —Ä—è–¥–∫–æ–≤–æ–≥–æ –ª—ñ—Ç–µ—Ä–∞–ª–∞, –≤–∫–∞–∑–∞–≤—à–∏ –∞–Ω–æ—Ç–∞—Ü—ñ—é —Ç–∏–ø—É `Character`:

```swift
let exclamationMark: Character = "!"
```

–ó–Ω–∞—á–µ–Ω–Ω—è `String` –º–æ–∂–Ω–∞ —Å–∫–æ–Ω—Å—Ç—Ä—É—é–≤–∞—Ç–∏, –ø–µ—Ä–µ–¥–∞–≤—à–∏ –º–∞—Å–∏–≤ —Å–∏–º–≤–æ–ª—ñ–≤ `Character` —è–∫ –∞—Ä–≥—É–º–µ–Ω—Ç –π–æ–≥–æ —ñ–Ω—ñ—Ü—ñ–∞–ª—ñ–∑–∞—Ç–æ—Ä–∞:

```swift
let catCharacters: [Character] = ["–ö", "–∏", "—Ü", "—è", "!", "üê±"]
let catString = String(catCharacters)
print(catString)
// –î—Ä—É–∫—É—î "–ö–∏—Ü—è!üê±"
```

## –ö–æ–Ω–∫–∞—Ç–µ–Ω–∞—Ü—ñ—è —Ä—è–¥–∫—ñ–≤ —Ç–∞ —Å–∏–º–≤–æ–ª—ñ–≤

–ó–Ω–∞—á–µ–Ω–Ω—è `String` –º–æ–∂–Ω–∞ –∑'—î–¥–Ω—É–≤–∞—Ç–∏ —Ä–∞–∑–æ–º \(–∞–±–æ _–∫–æ–Ω–∫–∞—Ç–µ–Ω—É–≤–∞—Ç–∏_\) –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é –æ–ø–µ—Ä–∞—Ç–æ—Ä–∞ –¥–æ–¥–∞–≤–∞–Ω–Ω—è \(`+`\), –ø—Ä–∏ —Ü—å–æ–º—É –±—É–¥–µ —Å—Ç–≤–æ—Ä–µ–Ω–æ –Ω–æ–≤–∏–π —Ä—è–¥–æ–∫:

```swift
let string1 = "hello"
let string2 = " there"
var welcome = string1 + string2
// –∑–º—ñ–Ω–Ω–∞ welcome —Ç–µ–ø–µ—Ä –¥–æ—Ä—ñ–≤–Ω—é—î "hello there"
```

–í–∏ –º–æ–∂–µ—Ç–µ —Ç–∞–∫–æ–∂ –¥–æ–¥–∞–≤–∞—Ç–∏ –∑–Ω–∞—á–µ–Ω–Ω—è `String` –≤ –∫—ñ–Ω–µ—Ü—å –Ω–∞—è–≤–Ω–æ—ó –∑–º—ñ–Ω–Ω–æ—ó `String` –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é –æ–ø–µ—Ä–∞—Ç–æ—Ä–∞ –¥–æ–¥–∞–≤–∞–Ω–Ω—è –∑ –ø—Ä–∏—Å–≤–æ—î–Ω–Ω—è–º \(`+=`\):

```swift
var instruction = "look over"
instruction += string2
// –∑–º—ñ–Ω–Ω–∞ instruction —Ç–µ–ø–µ—Ä –¥–æ—Ä—ñ–≤–Ω—é—î "look over there"
```

–í–∏ –º–æ–∂–µ—Ç–µ –¥–æ–¥–∞—Ç–∏ —Å–∏–º–≤–æ–ª `Character` –¥–æ –∑–º—ñ–Ω–Ω–æ—ó `String` –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é –º–µ—Ç–æ–¥—É `append()` —Ç–∏–ø—É `String`:

```swift
let exclamationMark: Character = "!"
welcome.append(exclamationMark)
// –∑–º—ñ–Ω–Ω–∞ welcome —Ç–µ–ø–µ—Ä –¥–æ—Ä—ñ–≤–Ω—é—î "hello there!"
```

> **–ü—Ä–∏–º—ñ—Ç–∫–∞**
>
> –í–∏ –Ω–µ –º–æ–∂–µ—Ç–µ –¥–æ–¥–∞—Ç–∏ —Ä—è–¥–æ–∫ `String` —á–∏ —Å–∏–º–≤–æ–ª `Character` –¥–æ —ñ—Å–Ω—É—é—á–æ—ó –∑–º—ñ–Ω–Ω–æ—ó `Character`, –±–æ –∑–Ω–∞—á–µ–Ω–Ω—è `Character` –º–æ–∂—É—Ç—å –º—ñ—Å—Ç–∏—Ç–∏ –ª–∏—à–µ –æ–¥–∏–Ω —Å–∏–º–≤–æ–ª.

## –Ü–Ω—Ç–µ—Ä–ø–æ–ª—è—Ü—ñ—è —Ä—è–¥–∫—ñ–≤

_–Ü–Ω—Ç–µ—Ä–ø–æ–ª—è—Ü—ñ—è —Ä—è–¥–∫—ñ–≤_ —î —Å–ø–æ—Å–æ–±–æ–º —Å—Ç–≤–æ—Ä–∏—Ç–∏ –Ω–æ–≤–∏–π —Ä—è–¥–æ–∫ `String` —ñ–∑ –∫–æ–º–±—ñ–Ω–∞—Ü—ñ—ó –∫–æ–Ω—Å—Ç–∞–Ω—Ç, –∑–º—ñ–Ω–Ω–∏—Ö, –ª—ñ—Ç–µ—Ä–∞–ª—ñ–≤ —Ç–∞ –≤–∏—Ä–∞–∑—ñ–≤, –≤–∫–ª—é—á–∞—é—á–∏ —ó—Ö –∑–Ω–∞—á–µ–Ω–Ω—è –≤—Å–µ—Ä–µ–¥–∏–Ω—ñ —Ä—è–¥–∫–æ–≤–æ–≥–æ –ª—ñ—Ç–µ—Ä–∞–ª–∞. –ö–æ–∂–µ–Ω –µ–ª–µ–º–µ–Ω—Ç, —â–æ –≤—Å—Ç–∞–≤–ª—è—î—Ç—å—Å—è –≤ —Ä—è–¥–∫–æ–≤–∏–π –ª—ñ—Ç–µ—Ä–∞–ª, –º–∞—î –±—É—Ç–∏ –æ—Ç–æ—á–µ–Ω–∏–π –ø–∞—Ä–æ—é –¥—É–∂–æ–∫, –ø–µ—Ä–µ–¥ —è–∫–∏–º–∏ –π–¥–µ –∑–≤–æ—Ä–æ—Ç–Ω–∏–π —Å–ª–µ—à \(`\`\):

```swift
let multiplier = 3
let message = "\(multiplier) —Ä–∞–∑–∏ –ø–æ 2.5 –¥–æ—Ä—ñ–≤–Ω—é—î \(Double(multiplier) * 2.5)"
// –∫–æ–Ω—Å—Ç–∞–Ω—Ç–∞ message –¥–æ—Ä—ñ–≤–Ω—é—î "3 —Ä–∞–∑–∏ –ø–æ 2.5 –¥–æ—Ä—ñ–≤–Ω—é—î–∏ 7.5"
```

–£ –ø—Ä–∏–∫–ª–∞–¥—ñ –≤–∏—â–µ, –∑–Ω–∞—á–µ–Ω–Ω—è `multiplier` –≤—Å—Ç–∞–≤–ª—è—î—Ç—å—Å—è –≤ —Ä—è–¥–∫–æ–≤–∏–π –ª—ñ—Ç–µ—Ä–∞–ª —è–∫ `\(multiplier)`. –¶–µ–π –∑–∞–ø–æ–≤–Ω—é–≤–∞—á –±—É–¥–µ –∑–∞–º—ñ–Ω–µ–Ω–æ —Ñ–∞–∫—Ç–∏—á–Ω–∏–º –∑–Ω–∞—á–µ–Ω–Ω—è–º –∫–æ–Ω—Å—Ç–∞–Ω—Ç–∏ `multiplier` –ø—ñ–¥ —á–∞—Å –≤–∏–∫–æ–Ω–∞–Ω–Ω—è —ñ–Ω—Ç–µ—Ä–ø–æ–ª—è—Ü—ñ—ó —Ä—è–¥–∫—É —É —Ö–æ–¥—ñ —Å—Ç–≤–æ—Ä–µ–Ω–Ω—è –¥–∞–Ω–æ–≥–æ —Ä—è–¥–∫–∞.

–ó–Ω–∞—á–µ–Ω–Ω—è `multiplier` —î —Ç–∞–∫–æ–∂ —á–∞—Å—Ç–∏–Ω–æ—é –±—ñ–ª—å—à–æ–≥–æ –≤–∏—Ä–∞–∑—É –¥–∞–ª—ñ –≤ —Ä—è–¥–∫—É. –¶–µ–π –≤–∏—Ä–∞–∑ –æ–±—á–∏—Å–ª—é—î –∑–Ω–∞—á–µ–Ω–Ω—è `Double(multiplier) * 2.5` —ñ –≤—Å—Ç–∞–≤–ª—è—î —Ä–µ–∑—É–ª—å—Ç–∞—Ç \(`7.5`\) –≤ —Ä—è–¥–æ–∫. –í —Ü—å–æ–º—É –≤–∏–ø–∞–¥–∫—É, –≤–∏—Ä–∞–∑ –∑–∞–ø–∏—Å—É—î—Ç—å—Å—è —è–∫ `\(Double(multiplier) * 2.5)` –¥–ª—è –≤–∫–ª—é—á–µ–Ω–Ω—è –≤ —Ä—è–¥–∫–æ–≤–∏–π –ª—ñ—Ç–µ—Ä–∞–ª.

> **–ü—Ä–∏–º—ñ—Ç–∫–∞**
>
> –í–∏—Ä–∞–∑–∏, –∫–æ—Ç—Ä—ñ –≤–∏ –ø–∏—à–µ—Ç–µ –≤—Å–µ—Ä–µ–¥–∏–Ω—ñ –¥—É–∂–æ–∫ –ø—Ä–∏ —ñ–Ω—Ç–µ—Ä–ø–æ–ª—è—Ü—ñ—ó —Ä—è–¥–∫—ñ–≤, –Ω–µ –º–æ–∂—É—Ç—å –º—ñ—Å—Ç–∏—Ç–∏ –Ω–µ–µ–∫—Ä–∞–Ω–æ–≤–∞–Ω–∏—Ö –∑–≤–æ—Ä–æ—Ç–Ω—ñ—Ö —Å–ª–µ—à—ñ–≤ \(`\`\), —Å–∏–º–≤–æ–ª—ñ–≤ –∑–º—ñ–Ω–∏ —Ä—è–¥–∫–∞ \(`\n`\) —á–∏ —Å–∏–º–≤–æ–ª—ñ–≤ –ø–æ–≤–µ—Ä–Ω–µ–Ω–Ω—è –∫–∞—Ä–µ—Ç–∫–∏ \(`\r`\). –û–¥–Ω–∞–∫ –≤–æ–Ω–∏ –º–æ–∂—É—Ç—å –º—ñ—Å—Ç–∏—Ç–∏ —ñ–Ω—à—ñ —Ä—è–¥–∫–æ–≤—ñ –ª—ñ—Ç–µ—Ä–∞–ª–∏.

## –Æ–Ω—ñ–∫–æ–¥

_–Æ–Ω—ñ–∫–æ–¥_ - —Ü–µ –º—ñ–∂–Ω–∞—Ä–æ–¥–Ω–∏–π —Å—Ç–∞–Ω–¥–∞—Ä—Ç –¥–ª—è –∫–æ–¥—É–≤–∞–Ω–Ω—è, –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–Ω—è —Ç–∞ –æ–±—Ä–æ–±–∫–∏ —Ç–µ–∫—Å—Ç—É –≤ —Ä—ñ–∑–Ω–∏—Ö —Å–∏—Å—Ç–µ–º–∞—Ö –ø–∏—Å–µ–º–Ω–æ—Å—Ç—ñ. –í—ñ–Ω –¥–æ–∑–≤–æ–ª—è—î –≤–∞–º –ø—Ä–µ–¥—Å—Ç–∞–≤–∏—Ç–∏ –ø—Ä–∞–∫—Ç–∏—á–Ω–æ –±—É–¥—å-—è–∫–∏–π —Å–∏–º–≤–æ–ª –∑ –±—É–¥—å-—è–∫–æ—ó –º–æ–≤–∏ —É —Å—Ç–∞–Ω–¥–∞—Ä—Ç–∏–∑–æ–≤–∞–Ω—ñ–π —Ñ–æ—Ä–º—ñ, –∑—á–∏—Ç—É–≤–∞—Ç–∏ —Ç–∞ –≤–∏–≤–æ–¥–∏—Ç–∏ —Ü—ñ —Å–∏–º–≤–æ–ª–∏ —ñ–∑ –±—É–¥—å-—è–∫–æ–≥–æ –∑–æ–≤–Ω—ñ—à–Ω—å–æ–≥–æ –¥–∂–µ—Ä–µ–ª–∞ –≤ –±—É–¥—å-—è–∫–µ —ñ–Ω—à–µ –∑–æ–≤–Ω—ñ—à–Ω—î –¥–∂–µ—Ä–µ–ª–æ, —Ç–∞–∫—ñ —è–∫ —Ç–µ–∫—Å—Ç–æ–≤–∏–π —Ñ–∞–π–ª —á–∏ –≤–µ–±-—Å—Ç–æ—Ä—ñ–Ω–∫–∞. –¢–∏–ø–∏ `String` —Ç–∞ `Character` —É –º–æ–≤—ñ Swift —î –ø–æ–≤–Ω—ñ—Å—Ç—é —Å—É–º—ñ—Å–Ω–∏–º–∏ —ñ–∑ –Æ–Ω—ñ–∫–æ–¥–æ–º, –ø—Ä–æ —â–æ –π —ñ—Ç–∏–º–µ –º–æ–≤–∞ —É –¥–∞–Ω–æ–º—É –ø—ñ–¥—Ä–æ–∑–¥—ñ–ª—ñ.

### –Æ–Ω—ñ–∫–æ–¥–æ–≤—ñ —Å–∫–∞–ª—è—Ä–∏

–ó–∞ –ª–∞—à—Ç—É–Ω–∫–∞–º–∏, –≤–±—É–¥–æ–≤–∞–Ω–∏–π —É Swift —Ç–∏–ø `String` –±—É–¥—É—î—Ç—å—Å—è —ñ–∑ —é–Ω—ñ–∫–æ–¥–æ–≤–∏—Ö —Å–∫–∞–ª—è—Ä–Ω–∏—Ö –∑–Ω–∞—á–µ–Ω—å. –Æ–Ω—ñ–∫–æ–¥–æ–≤–∏–º —Å–∫–∞–ª—è—Ä–æ–º —î —É–Ω—ñ–∫–∞–ª—å–Ω–µ 21-–±—ñ—Ç–Ω–µ —á–∏—Å–ª–æ –¥–ª—è –ø–æ–∑–Ω–∞—á–µ–Ω–Ω—è —Å–∏–º–≤–æ–ª—É —á–∏ –º–æ–¥–∏—Ñ—ñ–∫–∞—Ç–æ—Ä–∞, —è–∫, –Ω–∞–ø—Ä–∏–∫–ª–∞–¥, `U+0061` –¥–ª—è `LATIN SMALL LETTER A` \(`"a"`\), –∞–±–æ `U+1F425` –¥–ª—è `FRONT-FACING BABY CHICK` \(`"üê•"`\).

> **–ü—Ä–∏–º—ñ—Ç–∫–∞**
>
> –Æ–Ω—ñ–∫–æ–¥–æ–≤–∏–º —Å–∫–∞–ª—è—Ä–æ–º —î –±—É–¥—å-—è–∫–∞ _–∫–æ–¥–æ–≤–∞ –ø–æ–∑–∏—Ü—ñ—è_ —É –ø—Ä–æ–º—ñ–∂–∫—É –≤—ñ–¥ `U+0000` –¥–æ `U+D7FF` –≤–∫–ª—é—á–Ω–æ, –∞–±–æ –≤—ñ–¥ `U+E000` –¥–æ `U+10FFFF` –≤–∫–ª—é—á–Ω–æ. –Æ–Ω—ñ–∫–æ–¥–æ–≤—ñ —Å–∫–∞–ª—è—Ä–∏ –Ω–µ –≤–∫–ª—é—á–∞—é—Ç—å –∫–æ–¥–æ–≤—ñ –ø–æ–∑–∏—Ü—ñ—ó _—Å—É—Ä–æ–≥–∞—Ç–Ω–∏—Ö –ø–∞—Ä_, –∫–æ—Ç—Ä—ñ —î –∫–æ–¥–æ–≤–∏–º–∏ –ø–æ–∑–∏—Ü—ñ—è–º–∏ —É –ø—Ä–æ–º—ñ–∂–∫—É –≤—ñ–¥ `U+D800` –¥–æ `U+DFFF` –≤–∫–ª—é—á–Ω–æ.

### –Ü–Ω–¥–µ–∫—Å–∏ —Ä—è–¥–∫–∞

–Ø–∫ –∑–∞–∑–Ω–∞—á–µ–Ω–æ –≤–∏—â–µ, —Ä—ñ–∑–Ω—ñ —Å–∏–º–≤–æ–ª–∏ –º–æ–∂—É—Ç—å –≤–∏–º–∞–≥–∞—Ç–∏ —Ä—ñ–∑–Ω–æ—ó –∫—ñ–ª—å–∫–æ—Å—Ç—ñ –ø–∞–º'—è—Ç—ñ –¥–ª—è –∑–±–µ—Ä—ñ–≥–∞–Ω–Ω—è, —Ç–æ–º—É –¥–ª—è —Ç–æ–≥–æ, —â–æ–± –≤–∏–∑–Ω–∞—á–∏—Ç–∏, —è–∫–∏–π —Å–∞–º–µ —Å–∏–º–≤–æ–ª –∑–Ω–∞—Ö–æ–¥–∏—Ç—å—Å—è –∑–∞ –¥–∞–Ω–æ—é –ø–æ–∑–∏—Ü—ñ—î—é, —Å–ª—ñ–¥ –ø—Ä–æ—ñ—Ç–µ—Ä—É–≤–∞—Ç–∏ –∫–æ–∂–µ–Ω —é–Ω—ñ–∫–æ–¥–æ–≤–∏–π —Å–∫–∞–ª—è—Ä –∑ –ø–æ—á–∞—Ç–∫—É —á–∏ –∑ –∫—ñ–Ω—Ü—è —Ü—å–æ–≥–æ —Ä—è–¥–∫–∞. –ó —Ü—ñ—î—ó –ø—Ä–∏—á–∏–Ω–∏, —Ä—è–¥–∫—É —É Swift –Ω–µ –º–æ–∂—É—Ç—å –º–∞—Ç–∏ —Ü—ñ–ª–æ—á–∏—Å–µ–ª—å–Ω—ñ —ñ–Ω–¥–µ–∫—Å–∏.

–ó–∞ –¥–æ–ø–æ–º–æ–≥–æ—é –≤–ª–∞—Å—Ç–∏–≤–æ—Å—Ç—ñ `startIndex` –º–æ–∂–Ω–∞ –æ—Ç—Ä–∏–º–∞—Ç–∏ –ø–æ–∑–∏—Ü—ñ—é –ø–µ—Ä—à–æ–≥–æ —Å–∏–º–≤–æ–ª—É –≤ —Ä—è–¥–∫—É. –í–ª–∞—Å—Ç–∏–≤—ñ—Å—Ç—å `endIndex` —î –ø–æ–∑–∏—Ü—ñ—î—é –ø—ñ—Å–ª—è –æ—Å—Ç–∞–Ω–Ω—å–æ–≥–æ —Å–∏–º–≤–æ–ª—É –≤ —Ä—è–¥–∫—É. –Ø–∫ —Ä–µ–∑—É–ª—å—Ç–∞—Ç, –≤–ª–∞—Å—Ç–∏–≤—ñ—Å—Ç—å `endIndex` –Ω–µ —î –∫–æ—Ä–µ–∫—Ç–Ω–∏–º –∞—Ä–≥—É–º–µ–Ω—Ç–æ–º –¥–ª—è —ñ–Ω–¥–µ–∫—Å—É —Ä—è–¥–∫–∞. –Ø–∫—â–æ —Ä—è–¥–æ–∫ –ø–æ—Ä–æ–∂–Ω—ñ–π, –≤–ª–∞—Å—Ç–∏–≤–æ—Å—Ç—ñ `startIndex` —Ç–∞ `endIndex` –¥–æ—Ä—ñ–≤–Ω—é—é—Ç—å –æ–¥–Ω–∞ –æ–¥–Ω—ñ–π.

–©–æ–± –æ—Ç—Ä–∏–º–∞—Ç–∏ —ñ–Ω–¥–µ–∫—Å–∏ –ø–µ—Ä–µ–¥ —Ç–∞ –ø—ñ—Å–ª—è –¥–∞–Ω–æ–≥–æ —ñ–Ω–¥–µ–∫—Å—É, —Å–ª—ñ–¥ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É–≤–∞—Ç–∏ –º–µ—Ç–æ–¥–∏ `String` `index(before:)` —Ç–∞ `index(after:)` –≤—ñ–¥–ø–æ–≤—ñ–¥–Ω–æ. –©–æ–± –æ—Ç—Ä–∏–º–∞—Ç–∏ —ñ–Ω–¥–µ–∫—Å —Ç—Ä–æ—Ö–∏ –¥–∞–ª—ñ –≤—ñ–¥ –¥–∞–Ω–æ–≥–æ, —Å–ª—ñ–¥ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É–≤–∞—Ç–∏ –º–µ—Ç–æ–¥ `index(_:offsetBy:)` –∑–∞–º—ñ—Å—Ç—å —Ç–æ–≥–æ, —â–æ–± –≤–∏–∫–ª–∏–∫–∞—Ç–∏ –æ–¥–∏–Ω –∑ –ø–æ–ø–µ—Ä–µ–¥–Ω—ñ—Ö –º–µ—Ç–æ–¥—ñ–≤ –∫—ñ–ª—å–∫–∞ —Ä–∞–∑—ñ–≤.

–©–æ–± –æ—Ç—Ä–∏–º–∞—Ç–∏ —Å–∏–º–≤–æ–ª —Ä—è–¥–∫–∞ –∑–∞ –¥–∞–Ω–∏–º —ñ–Ω–¥–µ–∫—Å–æ–º, —Å–ª—ñ–¥ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É–≤–∞—Ç–∏ —Å–∏–Ω—Ç–∞–∫—Å–∏—Å —ñ–Ω–¥–µ–∫—Å–∞—Ü—ñ—ó \(`[]`\).

```swift
let greeting = "Guten Tag!"

// G
greeting[greeting.index(before: greeting.endIndex)]
// !
greeting[greeting.index(after: greeting.startIndex)]
// u

// a
```

–°–ø—Ä–æ–±–∏ –æ—Ç—Ä–∏–º–∞—Ç–∏ —ñ–Ω–¥–µ–∫—Å –∑–∞ –º–µ–∂–∞–º–∏ –¥—ñ–∞–ø–∞–∑–æ–Ω—É —Å–∏–º–≤–æ–ª—ñ–≤ —Ä—è–¥–∫–∞, —á–∏ –∑–≤–µ—Ä–Ω—É—Ç–∏—Å—å –¥–æ —Å–∏–º–≤–æ–ª—É –∑–∞ –º–µ–∂–∞–º–∏ —Ä—è–¥–∫–∞, –ø—Ä–∏–∑–≤–µ–¥–µ –¥–æ –ø–æ–º–∏–ª–∫–∏ –Ω–∞ –µ—Ç–∞–ø—ñ –≤–∏–∫–æ–Ω–∞–Ω–Ω—è.

> **–ü—Ä–∏–º—ñ—Ç–∫–∞**
>
> –ú–æ–∂–Ω–∞ –∫–æ—Ä–∏—Å—Ç—É–≤–∞—Ç–∏—Å—å –≤–ª–∞—Å—Ç–∏–≤–æ—Å—Ç—è–º–∏ `startIndex` —Ç–∞ `endIndex`, –∞ —Ç–∞–∫–æ–∂ –º–µ—Ç–æ–¥–∞–º–∏ `index(before:)`, `index(after:)`, —Ç–∞ `index(_:offsetBy:)` –Ω–∞ –±—É–¥—å-—è–∫–æ–º—É —Ç–∏–ø—ñ, —â–æ –ø—ñ–¥–ø–æ—Ä—è–¥–∫–æ–≤—É—î—Ç—å—Å—è –ø—Ä–æ—Ç–æ–∫–æ–ª—É `Collection`. –¢–∞–∫–∏–º —Ç–∏–ø–æ–º —î `String`, —è–∫ –ø–æ–∫–∞–∑–∞–Ω–æ —Ç—É—Ç, —Ç–∞–∫ —Å–∞–º–æ —Ç–∞–∫–∏–º–∏ —î —Ç–∏–ø–∏ –∫–æ–ª–µ–∫—Ü—ñ–π, —Ç–∞–∫—ñ —è–∫ `Array`, `Dictionary`, —Ç–∞ `Set`.

### –í—Å—Ç–∞–≤–∫–∞ —Ç–∞ –≤–∏–¥–∞–ª–µ–Ω–Ω—è

–©–æ–± –≤—Å—Ç–∞–≤–∏—Ç–∏ –æ–¥–∏–Ω —Å–∏–º–≤–æ–ª –≤ —Ä—è–¥–æ–∫ –∑–∞ –≤–∏–∑–Ω–∞—á–µ–Ω–∏–º —ñ–Ω–¥–µ–∫—Å–æ–º, –º–æ–∂–Ω–∞ –∫–æ—Ä–∏—Å—Ç—É–≤–∞—Ç–∏—Å—å –º–µ—Ç–æ–¥–æ–º `insert(_:at:)`, –∞ —â–æ–± –≤—Å—Ç–∞–≤–∏—Ç–∏ –≤–º—ñ—Å—Ç —ñ–Ω—à–æ–≥–æ —Ä—è–¥–∫–∞ –∑–∞ –≤–∏–∑–Ω–∞—á–µ–Ω–∏–º —ñ–Ω–¥–µ–∫—Å–æ–º, –º–æ–∂–Ω–∞ —Å–∫–æ—Ä–∏—Å—Ç–∞—Ç–∏—Å—å –º–µ—Ç–æ–¥–æ–º `insert(contentsOf:at:)`.

–©–æ–± –≤–∏–¥–∞–ª–∏—Ç–∏ –æ–¥–∏–Ω —Å–∏–º–≤–æ–ª —ñ–∑ —Ä—è–¥–∫–∞ –∑–∞ –≤–∏–∑–Ω–∞—á–µ–Ω–∏–º —ñ–Ω–¥–µ–∫—Å–æ–º, –º–æ–∂–Ω–∞ –≤–∏–∫–æ—Ä–∏—Å—Ç–∞—Ç–∏ –º–µ—Ç–æ–¥ `remove(at:)`, –∞ —â–æ–± –≤–∏–¥–∞–ª–∏—Ç–∏ –ø—ñ–¥—Ä—è–¥–æ–∫ –∑–∞ –≤–∏–∑–Ω–∞—á–µ–Ω–∏–º –¥—ñ–∞–ø–∞–∑–æ–Ω–æ–º, –º–æ–∂–Ω–∞ —Å–∫–æ—Ä–∏—Å—Ç–∞—Ç–∏—Å—å –º–µ—Ç–æ–¥–æ–º `removeSubrange(_:)`:

## –ü–æ—Ä—ñ–≤–Ω—è–Ω–Ω—è —Ä—è–¥–∫—ñ–≤

Swift –Ω–∞–¥–∞—î —Ç—Ä–∏ —Å–ø–æ—Å–æ–±–∏ –ø–æ—Ä—ñ–≤–Ω—é–≤–∞—Ç–∏ —Ç–µ–∫—Å—Ç–æ–≤—ñ –∑–Ω–∞—á–µ–Ω–Ω—è: —Ä—ñ–≤–Ω—ñ—Å—Ç—å —Ä—è–¥–∫—ñ–≤ —Ç–∞ —Å–∏–º–≤–æ–ª—ñ–≤, —Ä—ñ–≤–Ω—ñ—Å—Ç—å –ø—Ä–µ—Ñ—ñ–∫—Å—ñ–≤ —Ç–∞ —Ä—ñ–≤–Ω—ñ—Å—Ç—å —Å—É—Ñ—ñ–∫—Å—ñ–≤.

–î–≤–∞ —Ä—è–¥–∫–∏ `String` \(–∞–±–æ –¥–≤–∞ —Å–∏–º–≤–æ–ª–∏ `Character`\) –≤–≤–∞–∂–∞—é—Ç—å—Å—è –µ–∫–≤—ñ–≤–∞–ª–µ–Ω—Ç–Ω–∏–º–∏, —è–∫—â–æ —ó—Ö —Ä–æ–∑—à–∏—Ä–µ–Ω—ñ –∫–ª–∞—Å—Ç–µ—Ä–∏ –≥—Ä–∞—Ñ–µ–º —î _–∫–∞–Ω–æ–Ω—ñ—á–Ω–æ –µ–∫–≤—ñ–≤–∞–ª–µ–Ω—Ç–Ω—ñ_. –†–æ–∑—à–∏—Ä–µ–Ω—ñ –∫–ª–∞—Å—Ç–µ—Ä–∏ –≥—Ä–∞—Ñ–µ–º —î –∫–∞–Ω–æ–Ω—ñ—á–Ω–æ –µ–∫–≤—ñ–≤–∞–ª–µ–Ω—Ç–Ω–∏–º–∏, —è–∫—â–æ –≤–æ–Ω–∏ –º–∞—é—Ç—å –æ–¥–Ω–∞–∫–æ–≤—ñ –ª—ñ–Ω–≥–≤—ñ—Å—Ç–∏—á–Ω—ñ –∑–Ω–∞—á–µ–Ω–Ω—è —Ç–∞ –≤–∏–≥–ª—è–¥, –Ω–∞–≤—ñ—Ç—å —è–∫—â–æ –≤–æ–Ω–∏ —Å–∫–ª–∞–¥–∞—é—Ç—å—Å—è —ñ–∑ —Ä—ñ–∑–Ω–∏—Ö —é–Ω—ñ–∫–æ–¥–æ–≤–∏—Ö —Å–∫–∞–ª—è—Ä—ñ–≤ –∑–∞ –ª–∞—à—Ç—É–Ω–∫–∞–º–∏.

### –†—ñ–≤–Ω—ñ—Å—Ç—å –ø—Ä–µ—Ñ—ñ–∫—Å—ñ–≤ —Ç–∞ —Å—É—Ñ—ñ–∫—Å—ñ–≤

–©–æ–± –¥—ñ–∑–Ω–∞—Ç–∏—Å—å, —á–∏ –º–∞—î —Ä—è–¥–æ–∫ –ø–µ–≤–Ω–∏–π –ø—Ä–µ—Ñ—ñ–∫—Å —á–∏ —Å—É—Ñ—ñ–∫—Å, –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É–π—Ç–µ –º–µ—Ç–æ–¥–∏ `hasPrefix(_:)` —Ç–∞ `hasSuffix(_:)` –≤—ñ–¥–ø–æ–≤—ñ–¥–Ω–æ, –æ–±–∏–¥–≤–∞ –∑ —è–∫–∏—Ö –ø—Ä–∏–π–º–∞—é—Ç—å —î–¥–∏–Ω–∏–π –∞—Ä–≥—É–º–µ–Ω—Ç —Ç–∏–ø—É `String` —Ç–∞ –ø–æ–≤–µ—Ä—Ç–∞—é—Ç—å –±—É–ª–µ–≤–µ –∑–Ω–∞—á–µ–Ω–Ω—è.

–£ –ø—Ä–∏–∫–ª–∞–¥–∞—Ö –Ω–∏–∂—á–µ —Ä–æ–∑–≥–ª—è–¥–∞—î—Ç—å—Å—è –º–∞—Å–∏–≤ —Ä—è–¥–∫—ñ–≤, —â–æ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª—è—î –ª–æ–∫–∞—Ü—ñ—ó —Å—Ü–µ–Ω —ñ–∑ –ø–µ—Ä—à–∏—Ö –¥–≤–æ—Ö –∞–∫—Ç—ñ–≤ —à–µ–∫—Å–ø—ñ—Ä—ñ–≤—Å—å–∫–æ—ó _–†–æ–º–µ–æ —ñ –î–∂—É–ª—å—î—Ç—Ç–∞_:

–ú–æ–∂–Ω–∞ –≤–∏–∫–æ—Ä–∏—Å—Ç–∞—Ç–∏ –º–µ—Ç–æ–¥ `hasPrefix(_:)` –∑ –º–∞—Å–∏–≤–æ–º `romeoAndJuliet`, —â–æ–± –ø—ñ–¥—Ä–∞—Ö—É–≤–∞—Ç–∏ –∫—ñ–ª—å–∫—ñ—Å—Ç—å —Å—Ü–µ–Ω —É –ø–µ—Ä—à–æ–º—É –∞–∫—Ç—ñ –ø'—î—Å–∏:

```swift

    if scene.hasPrefix("Act 1 ") {

    }

// –ù–∞–¥—Ä—É–∫—É—î "–£ –ø–µ—Ä—à–æ–º—É –∞–∫—Ç—ñ 5 —Å—Ü–µ–Ω"
```

–ê–Ω–∞–ª–æ–≥—ñ—á–Ω–æ, –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é –º–µ—Ç–æ–¥—É `hasSuffix(_:)` –º–æ–∂–Ω–∞ –ø—ñ–¥—Ä–∞—Ö—É–≤–∞—Ç–∏ –∫—ñ–ª—å–∫—ñ—Å—Ç—å —Å—Ü–µ–Ω, —â–æ –≤—ñ–¥–±—É–≤–∞—î—Ç—å—Å—è –ø–æ–±–ª–∏–∑—É –ø–∞–ª–∞—Ü—É –ö–∞–ø—É–ª–µ—Ç—Ç—ñ \("Capulet‚Äôs mansion"\) —Ç–∞ –∫–µ–ª—ñ—ó –±—Ä–∞—Ç–∞ –õ–æ—Ä–µ–Ω—Ü–æ \(Friar Lawrence‚Äôs cell\):

>
> –ú–µ—Ç–æ–¥–∏ `hasPrefix(_:)` —Ç–∞ `hasSuffix(_:)` –≤–∏–∫–æ–Ω—É—é—Ç—å –ø–æ—Å–∏–º–≤–æ–ª—å–Ω—É –ø–µ—Ä–µ–≤—ñ—Ä–∫—É –Ω–∞ –∫–∞–Ω–æ–Ω—ñ—á–Ω—É –µ–∫–≤—ñ–≤–∞–ª–µ–Ω—Ç–Ω—ñ—Å—Ç—å –º—ñ–∂ —Ä–æ–∑—à–∏—Ä–µ–Ω–∏–º–∏ –∫–ª–∞—Å—Ç–µ—Ä–∞–º–∏ –≥—Ä–∞—Ñ–µ–º –≤ –∫–æ–∂–Ω–æ–º—É —Ä—è–¥–∫—É, —è–∫ –æ–ø–∏—Å–∞–Ω–æ –≤ [–†—ñ–≤–Ω—ñ—Å—Ç—å —Ä—è–¥–∫—ñ–≤ —Ç–∞ —Å–∏–º–≤–æ–ª—ñ–≤]({% link _book/1_language_guide/2_strings_and_characters.md %}#—Ä—ñ–≤–Ω—ñ—Å—Ç—å-—Ä—è–¥–∫—ñ–≤-—Ç–∞-—Å–∏–º–≤–æ–ª—ñ–≤).

## –Æ–Ω—ñ–∫–æ–¥–æ–≤–µ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–Ω—è —Ä—è–¥–∫—ñ–≤

–ö–æ–ª–∏ —é–Ω—ñ–∫–æ–¥–æ–≤–∏–π —Ä—è–¥–æ–∫ –∑–∞–ø–∏—Å—É—î—Ç—å—Å—è –≤ —Ç–µ–∫—Å—Ç–æ–≤–∏–π —Ñ–∞–π–ª —á–∏ –≤ —è–∫–µ—Å—å —ñ–Ω—à–µ –º—ñ—Å—Ü–µ –∑–±–µ—Ä—ñ–≥–∞–Ω–Ω—è, —é–Ω—ñ–∫–æ–¥–æ–≤—ñ —Å–∫–∞–ª—è—Ä–∏ –≤ —Ü—å–æ–º—É —Ä—è–¥–∫—É –∫–æ–¥—É—é—Ç—å—Å—è –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é –æ–¥–Ω–æ—ó —ñ–∑ _—Ñ–æ—Ä–º –∫–æ–¥—É–≤–∞–Ω–Ω—è_ –Æ–Ω—ñ–∫–æ–¥—É. –ö–æ–∂–Ω–∞ —Ñ–æ—Ä–º–∞ –∫–æ–¥—É—î —Ä—è–¥–æ–∫ –º–∞–ª–µ–Ω—å–∫–∏–º–∏ –ø–æ—Ä—Ü—ñ—è–º–∏, –∫–æ—Ç—Ä—ñ –≤—ñ–¥–æ–º—ñ —è–∫ _–æ–¥–∏–Ω–∏—Ü—ñ –∫–æ–¥—É–≤–∞–Ω–Ω—è_. –í–æ–Ω–∏ –≤–∫–ª—é—á–∞—é—Ç—å —Ñ–æ—Ä–º—É –∫–æ–¥—É–≤–∞–Ω–Ω—è UTF-8 \(–∫–æ—Ç—Ä–∞ –∫–æ–¥—É—î —Ä—è–¥–æ–∫ 8-–±—ñ—Ç–Ω–∏–º–∏ –æ–¥–∏–Ω–∏—Ü—è–º–∏ –∫–æ–¥—É–≤–∞–Ω–Ω—è\), —Ñ–æ—Ä–º—É –∫–æ–¥—É–≤–∞–Ω–Ω—è UTF-16 \(–∫–æ—Ç—Ä–∞ –∫–æ–¥—É—î —Ä—è–¥–æ–∫ 16-–±—ñ—Ç–Ω–∏–º–∏ –∫–æ–¥–æ–≤–∏–º–∏ –æ–¥–∏–Ω–∏—Ü—è–º–∏\), —Ç–∞ —Ñ–æ—Ä–º—É –∫–æ–¥—É–≤–∞–Ω–Ω—è UTF-32 \(–∫–æ—Ç—Ä–∞ –∫–æ–¥—É—î —Ä—è–¥–æ–∫ 32-–±—ñ—Ç–Ω–∏–º–∏ –∫–æ–¥–æ–≤–∏–º–∏ –æ–¥–∏–Ω–∏—Ü—è–º–∏\).

–ú–æ–≤–∞ Swift –Ω–∞–¥–∞—î –∫—ñ–ª—å–∫–∞ —Ä—ñ–∑–Ω–∏—Ö —Å–ø–æ—Å–æ–±—ñ–≤ –æ—Ç—Ä–∏–º–∞—Ç–∏ —é–Ω—ñ–∫–æ–¥–æ–≤—ñ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–Ω—è —Ä—è–¥–∫–∞. –ú–æ–∂–Ω–∞ —ñ—Ç–µ—Ä—É–≤–∞—Ç–∏ —Ä—è–¥–æ–∫ –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é —ñ–Ω—Å—Ç—Ä—É–∫—Ü—ñ—ó `for`-`in`, –æ—Ç—Ä–∏–º—É—é—á–∏ –æ–∫—Ä–µ–º—ñ –∑–Ω–∞—á–µ–Ω–Ω—è `Character` —è–∫ —é–Ω—ñ–∫–æ–¥–æ–≤—ñ —Ä–æ–∑—à–∏—Ä–µ–Ω—ñ –∫–ª–∞—Å—Ç–µ—Ä–∏ –≥—Ä–∞—Ñ–µ–º. –¶–µ–π –ø—Ä–æ—Ü–µ—Å –æ–ø–∏—Å–∞–Ω–æ —É –ø—ñ–¥—Ä–æ–∑–¥—ñ–ª—ñ [–†–æ–±–æ—Ç–∞ —ñ–∑ —Å–∏–º–≤–æ–ª–∞–º–∏]({% link _book/1_language_guide/2_strings_and_characters.md %}#—Ä–æ–±–æ—Ç–∞-—ñ–∑-—Å–∏–º–≤–æ–ª–∞–º–∏).

–Ü–Ω—à–∏–º —Å–ø–æ—Å–æ–±–æ–º —î –¥–æ—Å—Ç—É–ø –¥–æ —Ç—Ä—å–æ—Ö —ñ–Ω—à–∏—Ö –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω—å —Ä—è–¥–∫–∞ `String`:

* –∫–æ–ª–µ–∫—Ü—ñ—è –æ–¥–∏–Ω–∏—Ü—å –∫–æ–¥—É–≤–∞–Ω–Ω—è UTF-8 \(–∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é –≤–ª–∞—Å—Ç–∏–≤–æ—Å—Ç—ñ —Ä—è–¥–∫–∞ `utf8`\)
* –∫–æ–ª–µ–∫—Ü—ñ—è –æ–¥–∏–Ω–∏—Ü—å –∫–æ–¥—É–≤–∞–Ω–Ω—è UTF-16 \(–∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é –≤–ª–∞—Å—Ç–∏–≤–æ—Å—Ç—ñ —Ä—è–¥–∫–∞ `utf16`\)
* –∫–æ–ª–µ–∫—Ü—ñ—è 21-–±—ñ—Ç–Ω–∏—Ö —é–Ω—ñ–∫–æ–¥–æ–≤–∏—Ö —Å–∫–∞–ª—è—Ä—ñ–≤, –µ–∫–≤—ñ–≤–∞–ª–µ–Ω—Ç–Ω–∏—Ö —Ñ–æ—Ä–º—ñ –∫–æ–¥—É–≤–∞–Ω–Ω—è UTF-32 \(–∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é –≤–ª–∞—Å—Ç–∏–≤–æ—Å—Ç—ñ —Ä—è–¥–∫–∞ `unicodeScalars`\)

–ö–æ–∂–µ–Ω –ø—Ä–∏–∫–ª–∞–¥ –Ω–∏–∂—á–µ –¥–µ–º–æ–Ω—Å—Ç—Ä—É—î —Ä—ñ–∑–Ω—ñ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–Ω—è –Ω–∞—Å—Ç—É–ø–Ω–æ–≥–æ —Ä—è–¥–∫–∞, —â–æ —Å–∫–ª–∞–¥–∞—î—Ç—å—Å—è —ñ–∑ —Å–∏–º–≤–æ–ª—ñ–≤ `D`, `o`, `g`, `‚Äº` \(`DOUBLE EXCLAMATION MARK`, –∞–±–æ —é–Ω—ñ–∫–æ–¥–æ–≤–∏–π —Å–∫–∞–ª—è—Ä `U+203C`\) —Ç–∞ —Å–∏–º–≤–æ–ª—É `üê∂`\(`DOG FACE`, –∞–±–æ —é–Ω—ñ–∫–æ–¥–æ–≤–∏–π —Å–∫–∞–ª—è—Ä `U+1F436`\):


### –ü—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–Ω—è –≤ UTF-8

–û—Ç—Ä–∏–º–∞—î–º–æ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–Ω—è –≤ UTF-8 —Ä—è–¥–∫–∞ `String` —ñ—Ç–µ—Ä—É—é—á–∏ –π–æ–≥–æ –≤–ª–∞—Å—Ç–∏–≤—ñ—Å—Ç—å `utf8`. –¶—è –≤–ª–∞—Å—Ç–∏–≤—ñ—Å—Ç—å –º–∞—î —Ç–∏–ø `String.UTF8View`, –∫–æ—Ç—Ä–∏–π —î –∫–æ–ª–µ–∫—Ü—ñ—î—é –±–µ–∑–∑–Ω–∞–∫–æ–≤–∏—Ö 8-–±—ñ—Ç–Ω–∏—Ö –∑–Ω–∞—á–µ–Ω—å \(`UInt8`\), –ø–æ –æ–¥–Ω–æ–º—É –Ω–∞ –∫–æ–∂–µ–Ω –±–∞–π—Ç —É –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–Ω—è –≤ UTF-8:

–£ –ø—Ä–∏–∫–ª–∞–¥—ñ –≤–∏—â–µ, –ø–µ—Ä—à—ñ —Ç—Ä–∏ –¥–µ—Å—è—Ç–∫–æ–≤—ñ –∑–Ω–∞—á–µ–Ω–Ω—è `codeUnit` \(`68`, `111`, `103`\) –ø—Ä–µ–¥—Å—Ç–∞–≤–ª—è—é—Ç—å —Å–∏–º–≤–æ–ª–∏ `D`, `o`, —Ç–∞ `g`, —á–∏—î –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–Ω—è –≤ UTF-8 –∑–±—ñ–≥–∞—î—Ç—å—Å—è —ñ–∑ —ó—Ö –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–Ω—è–º —É –∫–æ–¥—É–≤–∞–Ω–Ω—ñ [ASCII](https://uk.wikipedia.org/wiki/ASCII). –ù–∞—Å—Ç—É–ø–Ω—ñ —Ç—Ä–∏ –¥–µ—Å—è—Ç–∫–æ–≤—ñ –∑–Ω–∞—á–µ–Ω–Ω—è `codeUnit` \(`226`, `128`, `188`\) —î —Ç—Ä—å–æ—Ö –±–∞–π—Ç–æ–≤–∏–º –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–Ω—è–º –≤ UTF-8 —Å–∏–º–≤–æ–ª—É `DOUBLE EXCLAMATION MARK`. –û—Å—Ç–∞–Ω–Ω—ñ —á–æ—Ç–∏—Ä–∏ –∑–Ω–∞—á–µ–Ω–Ω—è `codeUnit` \(`240`, `159`, `144`, `182`\) —î —á–æ—Ç–∏—Ä—å–æ—Ö –±–∞–π—Ç–æ–≤–∏–º –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–Ω—è–º –≤ UTF-8 —Å–∏–º–≤–æ–ª—É `DOG FACE`.

### –ü—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–Ω—è –≤ UTF-16

–û—Ç—Ä–∏–º–∞—î–º–æ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–Ω—è –≤ UTF-16 —Ä—è–¥–∫–∞ `String` —ñ—Ç–µ—Ä—É—é—á–∏ –π–æ–≥–æ –≤–ª–∞—Å—Ç–∏–≤—ñ—Å—Ç—å `utf16`. –¶—è –≤–ª–∞—Å—Ç–∏–≤—ñ—Å—Ç—å –º–∞—î —Ç–∏–ø `String.UTF16View`, –∫–æ—Ç—Ä–∏–π —î –∫–æ–ª–µ–∫—Ü—ñ—î—é –±–µ–∑–∑–Ω–∞–∫–æ–≤–∏—Ö 16-–±—ñ—Ç–Ω–∏—Ö –∑–Ω–∞—á–µ–Ω—å \(`UInt16`\), –ø–æ –æ–¥–Ω–æ–º—É –Ω–∞ –∫–æ–∂–Ω—É 16-–±—ñ—Ç–Ω—É –æ–¥–∏–Ω–∏—Ü—é –∫–æ–¥—É–≤–∞–Ω–Ω—è —É –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–Ω—ñ –≤ UTF-16:

–ó–Ω–æ–≤—É, –ø–µ—Ä—à—ñ —Ç—Ä–∏ –¥–µ—Å—è—Ç–∫–æ–≤—ñ –∑–Ω–∞—á–µ–Ω–Ω—è `codeUnit` \(`68`, `111`, `103`\) –ø—Ä–µ–¥—Å—Ç–∞–≤–ª—è—é—Ç—å —Å–∏–º–≤–æ–ª–∏ `D`, `o`, —Ç–∞ `g`, —á–∏—î –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–Ω—è –≤ –∫–æ–¥–æ–≤–∏—Ö –æ–¥–∏–Ω–∏—Ü—è—Ö UTF-16 –∑–±—ñ–≥–∞—î—Ç—å—Å—è —ñ–∑ —ó—Ö –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–Ω—è–º —É –∫–æ–¥—É–≤–∞–Ω–Ω—ñ UTF-8 \(–±–æ —Ü—ñ —é–Ω—ñ–∫–æ–¥–æ–≤—ñ —Å–∫–∞–ª—è—Ä–∏ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª—è—é—Ç—å —Å–∏–º–≤–æ–ª–∏ ASCII\).

–ß–µ—Ç–≤–µ—Ä—Ç–∏–º –∑–Ω–∞—á–µ–Ω–Ω—è–º `codeUnit` \(`8252`\) —î –¥–µ—Å—è—Ç–∫–æ–≤–∏–π –µ–∫–≤—ñ–≤–∞–ª–µ–Ω—Ç —à—ñ—Å—Ç–Ω–∞–¥—Ü—è—Ç–∫–æ–≤–æ–≥–æ –∑–Ω–∞—á–µ–Ω–Ω—è `203C`, —â–æ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª—è—î —é–Ω—ñ–∫–æ–¥–æ–≤–∏–π —Å–∫–∞–ª—è—Ä `U+203C` –¥–ª—è —Å–∏–º–≤–æ–ª—É `DOUBLE EXCLAMATION MARK`. –¶–µ–π —Å–∏–º–≤–æ–ª –º–æ–∂–µ –±—É—Ç–∏ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–æ —è–∫ —î–¥–∏–Ω–∞ –æ–¥–∏–Ω–∏—Ü—è –∫–æ–¥—É–≤–∞–Ω–Ω—è –≤ UTF-16.

–ü'—è—Ç–µ —ñ —à–æ—Å—Ç–µ –∑–Ω–∞—á–µ–Ω–Ω—è `codeUnit` \(`55357` —Ç–∞ `56374`\) —î –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–Ω—è–º —É –≤–∏–≥–ª—è–¥—ñ —Å—É—Ä–æ–≥–∞—Ç–Ω–æ—ó –ø–∞—Ä–∏ UTF-16 —Å–∏–º–≤–æ–ª—É `DOG FACE`. –¶—ñ –∑–Ω–∞—á–µ–Ω–Ω—è —î –≤–∏—â–∏–º —Å—É—Ä–æ–≥–∞—Ç–æ–º `U+D83D` \(—ñ–∑ –¥–µ—Å—è—Ç–∫–æ–≤–∏–º –∑–Ω–∞—á–µ–Ω–Ω—è–º `55357`\) —Ç–∞ –Ω–∏–∂—á–∏–º —Å—É—Ä–æ–≥–∞—Ç–æ–º `U+DC36` \(—ñ–∑ –¥–µ—Å—è—Ç–∫–æ–≤–∏–º –∑–Ω–∞—á–µ–Ω–Ω—è–º `56374`\).
